# 中国东北三省地图示例

这是从社区获取的地图示例

```python
from datetime import date
import math

import cartopy.crs as ccrs
import cartopy.feature as cfeature
import matplotlib.colors as mcolors
import matplotlib.pyplot as plt
import numpy as np
import rasterio
from shapely.ops import polylabel
from cartopy.io.shapereader import Reader

import frykit
import frykit.plot as fplt


ROOT = ""
SLOPE_TIF = f""
PROVINCE_SHP = f""
BOUNDARY_SHP = f""
OUTPUT_PNG = f""


def main() -> None:
    frykit.config.fast_transform = False
    frykit.config.strict_clip = True

    plt.rcParams.update(
        {
            "font.family": "sans-serif",
            "font.sans-serif": ["Songti SC", "STSong", "SimSun", "Noto Serif CJK SC", "Arial Unicode MS", "DejaVu Sans"],
            "axes.unicode_minus": False,
        }
    )

    map_crs = ccrs.LambertConformal(
        central_longitude=127,
        central_latitude=46,
        standard_parallels=(41, 49),
    )
    data_crs = ccrs.AlbersEqualArea(
        central_longitude=105,
        central_latitude=30,
        standard_parallels=(25, 47),
    )
    geo_crs = ccrs.PlateCarree()

    with rasterio.open(SLOPE_TIF) as src:
        slope = src.read(1)
        nodata = src.nodata
        bounds = src.bounds

    slope = np.ma.masked_equal(slope, nodata)

    province_records = list(Reader(PROVINCE_SHP).records())
    province_geoms = [record.geometry for record in province_records]
    boundary_geom = list(Reader(BOUNDARY_SHP).geometries())[0]

    xmin = min(geom.bounds[0] for geom in province_geoms)
    ymin = min(geom.bounds[1] for geom in province_geoms)
    xmax = max(geom.bounds[2] for geom in province_geoms)
    ymax = max(geom.bounds[3] for geom in province_geoms)
    map_extent = (xmin - 0.5, xmax + 0.5, ymin - 0.4, ymax + 0.5)
    xticks = list(range(math.floor(map_extent[0]), math.ceil(map_extent[1]) + 1))
    yticks = list(range(math.floor(map_extent[2]), math.ceil(map_extent[3]) + 1))

    levels = [0, 1, 2, 3, 5, 8, 12, 18]
    colors = [
        "#f7fcf0",
        "#d9f0c1",
        "#addd8e",
        "#78c679",
        "#41ab5d",
        "#238443",
        "#6b3f1d",
    ]
    cmap = mcolors.ListedColormap(colors)
    norm = mcolors.BoundaryNorm(levels, ncolors=cmap.N, clip=True)

    fig = plt.figure(figsize=(9.6, 10.8), facecolor="none")
    ax = fig.add_subplot(1, 1, 1, projection=map_crs)
    ax.set_facecolor("none")
    ax.set_extent(map_extent, crs=geo_crs)

    ax.add_feature(cfeature.OCEAN.with_scale("50m"), facecolor="#dceaf4", edgecolor="none", zorder=0)
    ax.add_feature(cfeature.LAND.with_scale("50m"), facecolor="#f3f0e8", edgecolor="none", zorder=0)
    ax.add_feature(cfeature.LAKES.with_scale("50m"), facecolor="#dceaf4", edgecolor="#9dbed3", linewidth=0.4, zorder=0.2)
    ax.add_feature(cfeature.RIVERS.with_scale("50m"), edgecolor="#9dbed3", linewidth=0.35, zorder=0.25)
    ax.add_feature(cfeature.COASTLINE.with_scale("50m"), edgecolor="#7f8b96", linewidth=0.5, zorder=0.3)
    ax.add_feature(cfeature.BORDERS.with_scale("50m"), edgecolor="#97a2ad", linewidth=0.45, linestyle="-", zorder=0.3)

    fplt.set_map_ticks(
        ax,
        extents=map_extent,
        xticks=xticks,
        yticks=yticks,
    )
    ax.tick_params(which="major", length=4.5, width=0.7, labelsize=8, direction="in")
    ax.gridlines(
        xlocs=xticks,
        ylocs=yticks,
        linewidth=0.45,
        linestyle="--",
        color="#8e98a3",
        alpha=0.45,
    )

    image = ax.imshow(
        slope,
        extent=(bounds.left, bounds.right, bounds.bottom, bounds.top),
        origin="upper",
        transform=data_crs,
        cmap=cmap,
        norm=norm,
        interpolation="nearest",
        zorder=1,
    )
    fplt.clip_by_polygon(image, boundary_geom, strict_clip=True)

    fplt.add_geometries(ax, province_geoms, fc="none", ec="#4a4a4a", lw=0.75, zorder=3)
    fplt.add_geometries(ax, boundary_geom, fc="none", ec="#1f1f1f", lw=1.2, crs=data_crs, zorder=4)

    label_offsets = {
        "辽宁省": (0.0, -0.12),
        "吉林省": (0.05, 0.05),
        "黑龙江省": (0.0, 0.0),
    }
    for record in province_records:
        geom = record.geometry
        if geom.geom_type == "MultiPolygon":
            geom = max(geom.geoms, key=lambda g: g.area)
        point = polylabel(geom, tolerance=0.01)
        dx, dy = label_offsets.get(record.attributes["name"], (0.0, 0.0))
        ax.text(
            point.x + dx,
            point.y + dy,
            record.attributes["name"],
            transform=geo_crs,
            ha="center",
            va="center",
            fontsize=12,
            fontweight="bold",
            color="#1f1f1f",
            zorder=5,
        )

    cax = fplt.add_side_axes(ax, width=0.032, pad=0.028, loc="right")
    cbar = plt.colorbar(image, cax=cax, ticks=levels, spacing="proportional")
    cbar.ax.tick_params(labelsize=9, length=3)
    cbar.set_label("坡度（°）", fontsize=11, labelpad=10)

    scale_bar = fplt.add_scale_bar(ax, x=0.62, y=0.035, length=500)
    scale_bar.set_xticks([0, 250, 500])
    scale_bar.set_xticklabels(["0", "250", "500 km"])
    scale_bar.tick_params(axis="x", labelsize=8, length=0, pad=2)
    scale_bar.set_yticks([])
    scale_bar.set_xlabel("")
    scale_bar.xaxis.set_ticks_position("bottom")
    for spine in scale_bar.spines.values():
        spine.set_visible(False)
    fplt.add_frame(scale_bar)

    fplt.add_compass(
        ax,
        x=0.92,
        y=0.92,
        size=24,
        style="arrow",
        pc_kwargs={"linewidth": 1.0, "edgecolor": "#1f1f1f"},
        text_kwargs={"fontsize": 11, "fontweight": "bold", "color": "#1f1f1f"},
    )

    ax.set_title("东北三省坡度地图", fontsize=18, fontweight="bold", pad=18)
    ax.text(
        0.0,
        -0.035,
        f"制图：GeoAgent | {date.today().isoformat()}",
        transform=ax.transAxes,
        ha="left",
        va="top",
        fontsize=8,
        color="#6f7780",
    )

    plt.savefig(OUTPUT_PNG, dpi=300, bbox_inches="tight", transparent=True)
    plt.close(fig)


if __name__ == "__main__":
    main()
```
